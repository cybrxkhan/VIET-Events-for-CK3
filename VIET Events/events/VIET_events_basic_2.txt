# Local woman accuses another of throwing strawberries at her while she was pregnant
VIETmisc.0400 = {
	type = character_event
	title = VIETmisc.0400.t
	desc = VIETmisc.0400.desc
	theme = VIET_theme_small_town


	right_portrait = {
		character = root
		animation = boredom
	}

	trigger = {	
		has_game_rule = VIET_normal_universe_events
		capital_province = {
			VIET_is_in_wild_strawberry_region_trigger = yes
		}	
		# This appears to be a medieval (Western?) European superstition
		culture_has_european_heritage_pillar_trigger = yes
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = compassionate
		}
		modifier = {
			add = -0.2
			has_trait = callous
		}
		modifier = {
			add = 0.2
			has_trait = just
		}
		modifier = {
			add = -0.2
			has_trait = arbitrary
		}
		modifier = {
			add = 0.2
			has_trait = patient
		}
		modifier = {
			add = -0.2
			has_trait = impatient
		}
	}
	
	option = { # Support the mother
		name = VIETmisc.0400.a
		add_character_modifier = {
			modifier = VIET_sympathetic_to_mothers
			years = 5
		}
	}	
	
	option = { # Support the other woman
		name = VIETmisc.0400.b
		add_character_modifier = {
			modifier = VIET_suspicious_of_hystical_women
			years = 5
		}
	}	
	
	option = { # GTFO
		name = VIETmisc.0400.c
		stress_impact = {
			just = miniscule_stress_impact_gain
			patient = miniscule_stress_impact_gain
			calm = miniscule_stress_impact_gain
		}
	}	


	after = {		
		if = {
			limit = {
				has_game_rule = VIET_historical_context_on
			}
			custom_tooltip = VIETmisc.0400.info
		}
	}

}



# Inspecting a church under construction with chaplain, and seeing the strawberries being drawn everywhere
VIETmisc.0401 = {
	type = character_event
	title = VIETmisc.0401.t
	desc = VIETmisc.0401.desc
	theme = VIET_theme_small_town


	left_portrait = {
		character = root
		animation = personality_zealous
	}
	
	right_portrait = {
		character = scope:chaplain
		animation = storyteller
	}

	trigger = {	
		has_game_rule = VIET_normal_universe_events
		exists = cp:councillor_court_chaplain
		cp:councillor_court_chaplain = {
			is_available_ai_adult = yes
			religion = religion:christianity_religion
		}
		religion = religion:christianity_religion
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = zealous
		}
		modifier = {
			add = -0.2
			has_trait = cynical
		}
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = diligent
		}
		modifier = {
			add = -0.2
			has_trait = lazy
		}
	}
	
	immediate = {
		cp:councillor_court_chaplain = {
			save_scope_as = chaplain
		}	
	}

	option = { # like it
		name = VIETmisc.0401.a
		VIET_medium_prestige_gain_effect = yes
		scope:chaplain = {			
			add_opinion = {
				modifier = pleased_opinion
				target = root
				opinion = 10
			}				
		}
		stress_impact = {
			zealous = miniscule_stress_impact_loss
			humble = miniscule_stress_impact_loss
		}
	}	

	option = { # don't like it
		name = VIETmisc.0401.a
		VIET_medium_piety_gain_effect = yes
		scope:chaplain = {			
			add_opinion = {
				modifier = disappointed_opinion
				opinion = -10
				target = root
			}				
		}
		stress_impact = {
			honest = miniscule_stress_impact_loss
			temperate = miniscule_stress_impact_loss
		}
	}	


	after = {		
		if = {
			limit = {
				has_game_rule = VIET_historical_context_on
			}
			custom_tooltip = VIETmisc.0401.info
		}
	}

}
