# Abacus - Merchant offers you his old abacus
VIETmisc.5000 = {
	type = character_event
	title = VIETmisc.5000.t
	desc = VIETmisc.5000.desc
	theme = stewardship
	override_background = { reference = relaxing_room_location }
	
	right_portrait = {
		character = root
		animation = personality_forgiving
	}
	artifact = { # To display the artifact in the event-window
		target = scope:VIET_clutter_artifact
		position = lower_center_portrait
	}
	left_portrait = {
		character = scope:merchant
		animation = personality_content
	}
	
	weight_multiplier = {
		base = 1
		# Reduce spam
		modifier = {
			add = -0.9
			any_character_artifact = {
				has_variable = VIET_abacus_variable
			}
		}
		modifier = {
			add = -0.4
			VIET_has_2_clutter_artifacts_trigger = yes
		}
		modifier = {
			add = -0.5
			VIET_has_4_clutter_artifacts_trigger = yes
		}
		modifier = {
			add = 0.2
			has_trait = diligent
		}
		modifier = {
			add = -0.2
			has_trait = lazy
		}
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = greedy
		}
		modifier = {
			add = -0.2
			has_trait = callous
		}
	}
	
	immediate = {
		
		VIET_find_old_merchant_effect = yes
		
		root = { add_visiting_courtier = scope:merchant }
		
		hidden_effect = {
			scope:merchant = {
				add_gold = medium_gold_value
				create_artifact = {
					name = VIET_abacus_name
					description = VIET_abacus_desc
					type = VIET_clutter
					visuals = VIET_abacus
					quality = 15
					wealth = 15
					modifier = artifact_VIET_abacus_modifier
					max_durability = 40
					save_scope_as = VIET_clutter_artifact
					history = {
						type = created_before_history
					}
				}
				scope:VIET_clutter_artifact = {
					set_variable = VIET_abacus_variable
				}
			}
		}
	}
	
	option = {
		name = VIETmisc.5000.a
		trigger = {
			OR = {
				is_ai = no
				gold >= 20
			}
		}
		pay_short_term_gold = {
			target = scope:merchant
			gold = 20
		}
		scope:VIET_clutter_artifact = {
			set_owner = {
				target = root
				history = {
					type = purchased
					actor = scope:merchant
					recipient = root
				}
			}
		}
		stress_impact = {
			arrogant = miniscule_stress_impact_gain
			lazy = miniscule_stress_impact_gain
		}
		custom_tooltip = VIET_artifact_limit_warning
	}
	
	option = {
		name = VIETmisc.5000.b
		VIET_medium_prestige_gain_effect = yes
		hidden_effect = {
			destroy_artifact = scope:VIET_clutter_artifact
		}
		stress_impact = {
			diligent = miniscule_stress_impact_gain
		}
	}
}

		}

		animation = personality_compassionate
	}
	artifact = { # To display the artifact in the event-window
		target = scope:VIET_clutter_artifact
		position = lower_center_portrait
	}
	
	weight_multiplier = {
		base = 1
		# Reduce spam
		modifier = {
			add = -0.9
			any_character_artifact = {
				has_variable = VIET_nutcracker_variable
			}
		}
		modifier = {
			add = -0.4
			VIET_has_2_clutter_artifacts_trigger = yes
		}
		modifier = {
			add = -0.5
			VIET_has_4_clutter_artifacts_trigger = yes
		}
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = compassionate
		}
		modifier = {
			add = -0.2
			has_trait = sadistic
		}
		modifier = {
			add = 0.2
			has_trait = generous
		}
		modifier = {
			add = -0.2
			has_trait = greedy
		}
	}
	
	trigger = {
		any_courtier = {
			is_available_ai_adult = yes
		}
	}
	
	immediate = {
		save_scope_as = likeness
		random_courtier = {
			limit = {
				is_available_ai_adult = yes
			}
			save_scope_as = nutgifter
			save_scope_as = owner # For decorations
		}
		hidden_effect = {
			scope:nutgifter = {
				create_artifact = {
					name = VIET_nutcracker_name
					description = placeholder
					type = VIET_clutter
					visuals = VIET_nutcracker
					quality = 15
					wealth = 15
					modifier = artifact_VIET_nutcracker_modifier
					max_durability = 40
					save_scope_as = VIET_clutter_artifact
					history = {
						type = created_before_history
					}
				}
			}
			scope:VIET_clutter_artifact = {
				set_variable = VIET_nutcracker_variable
				VIET_set_artifact_wealth_med_effect = yes
				set_artifact_feature_group = generic_material_wood
				if = {
					limit = { root = { faith_is_aniconic_trigger = yes } }
					set_artifact_description = VIET_nutcracker_aniconic_desc
				}
				else = {
					set_artifact_description = VIET_nutcracker_desc
				}
			}
		}
	}
	
	option = {
		name = VIETmisc.5004.a
		VIET_medium_prestige_loss_effect = yes
		scope:VIET_clutter_artifact = { set_owner = { target = root } }
		stress_impact = {
			paranoid = minor_stress_impact_gain
			arrogant = miniscule_stress_impact_gain
			callous = miniscule_stress_impact_gain
		}
		scope:nutgifter = {
			add_opinion = {
				target = root
				modifier = VIET_opinion_accepted_my_gift
			}
			reverse_add_opinion = {
				target = root
				modifier = VIET_opinion_gave_me_a_gift
			}
		}
		custom_tooltip = VIET_artifact_limit_warning
	}
	
	option = {
		name = VIETmisc.5004.b
		scope:nutgifter = {
			add_opinion = {
				target = root
				modifier = VIET_opinion_rejected_my_gift
			}
		}
		VIET_medium_prestige_gain_effect = yes
		hidden_effect = {
			destroy_artifact = scope:VIET_clutter_artifact
		}
		stress_impact = {
			compassionate = miniscule_stress_impact_gain
		}
	}
}
