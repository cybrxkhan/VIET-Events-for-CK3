# Abacus - Merchant offers you his old abacus
VIETmisc.5000 = {
	type = character_event
	title = VIETmisc.5000.t
	desc = VIETmisc.5000.desc
	theme = stewardship
	override_background = { event_background = council_chamber }
	
	left_portrait = root
	artifact = { # To display the artifact in the event-window
        target = scope:VIET_clutter_artifact
        position = lower_center_portrait
    }

    weight_multiplier = {
        base = 1
		# Reduce spam
		modifier = {
			add = -0.9
			any_character_artifact = {
				has_variable = VIET_abacus_variable
			}
		}
		modifier = {
			add = -0.4
			VIET_has_2_clutter_artifacts_trigger = yes	
		}
		modifier = {
			add = -0.5
			VIET_has_4_clutter_artifacts_trigger = yes	
		}
		modifier = {
			add = 0.2
			has_trait = diligent
		}
		modifier = {
			add = -0.2
			has_trait = lazy
		}
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = greedy
		}
		modifier = {
			add = -0.2
			has_trait = callous
		}
    }
	
	immediate = {	
		hidden_effect_new_artifact = {
			create_artifact = {	
				name = VIET_abacus_name
				description = VIET_abacus_desc
				type = VIET_clutter
				visuals = VIET_abacus
				quality = 15
				wealth = 15
				modifier = artifact_VIET_abacus_modifier
				max_durability = 40
				save_scope_as = VIET_clutter_artifact
				history = {
					type = created_before_history
				}
			}
			scope:VIET_clutter_artifact = {
				set_variable = VIET_abacus_variable
			}
		}
	}
	
	option = {
		name = VIETmisc.5000.a
		show_as_tooltip = {
			scope:VIET_clutter_artifact = {
				set_owner = {
					target = root
				}
			}
		}
		stress_impact = {
			arrogant = miniscule_stress_impact_gain
			lazy = miniscule_stress_impact_gain
		}
		custom_tooltip = VIET_artifact_limit_warning
	}
	
	option = {
		name = VIETmisc.5000.b
		VIET_medium_prestige_gain_effect = yes
		hidden_effect = {
			destroy_artifact = scope:VIET_clutter_artifact
		}
		stress_impact = {
			diligent = miniscule_stress_impact_gain
		}
	}
}
