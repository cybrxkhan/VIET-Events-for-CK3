# Arrival in any province during travel
on_travel_plan_movement = {
	on_actions = {
		delay = { days = { 2 28 } }
		VIET_travel_event_tombola		
	}

}


VIET_travel_event_tombola = {
	random_on_action = {
		6 = VIET_travel_events_common
		4 = VIET_travel_events_uncommon
		2 = VIET_travel_events_rare
	}
}
